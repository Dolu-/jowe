 89 /**
 90  * Creates a new random height map generator object.
 91  * @class Represents an HeightMap.
 92  * @see How to use the <a href="http://jowe.ouebfrance.com/examples-jowe-core.html">HeightMap</a> object?
 93  * @param {number} [opt_pitch=8]    Maximum elevation to be used. Items in the map will have a value between 0 and opt_pitch.
 94  * @param {number} [opt_ratio=3.1]  Ratio to be used.
 95  * @param {number} [opt_width=127]  Width of the map to be generated.
 96  * @param {number} [opt_height=127] Height of the map to be generated.
 97  */ 
 98 function HeightMap(opt_pitch, opt_ratio, opt_width, opt_height) {
 99     
100     /**
101      * Set minimum size for the side value of the working square.
102      * @type number
103      * @private
104      * @constant
105      */
106     var MIN_SIDE = 4;
107     
108     /**
109      * Set maximum size for the side value of the working square.
110      * @type number
111      * @private
112      * @constant
113      */
114     var MAX_SIDE = 1010;
115 
116     /**
117      * Borders of the working height map to exclude from final result,
118      * because there are not processed in the smooth function.
119      * @type number
120      * @private
121      * @constant
122      */
123     var CROP_SIZE = 1;
124         
125     /**
126      * Maximum elevation for current map [0 to p_pitch].
127      * <br />
128      * You will have to adjust the color managment in "jowe-ui" to fit your elevation.
129      * @type number
130      * @private
131      */
132     var p_pitch = 8;
133         
134     /**
135      * Indicates how much height difference between 2 points we can have.
136      * Only used in "make" method.
137      * <br />
138      * By the way, combined with "p_pitch" (previous property),
139      * it allows to obtain very different types of map.
140      * <br/>
141      * For now 3.1 is around the minimum value to use, below that you
142      * could obtain strange map (unmanaged cell display).
143      * If you put an higher value, your map will look flattened.
144      * @type number
145      * @private
146      */
147     var p_ratio = 3.1;
148 
149     /**
150      * Size of the current map [0 to p_side], [0 to p_side].
151      * <br />
152      * Always have to be (N^2)+1 x (N^2)+1 ("diamond square" algorithm. Better looking results with squares).
153      * We only need to store it once (p_side = width = height).
154      * @type number
155      * @private
156      */
157     var p_side = 129,
158         
159         // "real" dimension of the current object (not resized to 2^n square) and not cropped.
160         width = 127,
161         height = 127,
162         
163         // Random object/class.
164         rand;
165 
166     /**
167      * Shorcut to Math function.
168      * <br />
169      * Used to "speed up" calls to Math.floor() function.
170      * @function
171      * @private
172      */
173     var floor = Math.floor;
174     
175     /**
176      * Result array with the world map.
177      * @type Array
178      * @public
179      */
180     this.item = [];
181     
182     /**
183      * Initialize the item array before calling the generator method.
184      * The whole item array is filled with the default value given as parameter.
185      * @param {number} h default value given to each item.
186      * @public
187      */
188     this.initialize = function (h) {
189         var x = p_side, a = this.item, b = [];
190         a.length = b.length = p_side;
191         while (x) b[--x] = h;
192         x = p_side;
193         while (x) a[--x] = b.slice();
194     };
195         
196     /**
197      * Generate a random value between min and max (both included), result is floored.
198      * @param {number} min
199      * @param {number} max
200      * @return {number} Random value.
201      * @private
202      */
203     function p_randomMinMax(min, max) {
204         return floor((rand() * ((max - min) + 1)) + min);
205     }
206 
207     /**
208      * Set random value for each corner, needed before calling the generator.
209      * <br />
210      * Require the item array to be initialized.
211      * @param {boolean} overwrite Indicates if existing values should be overwritten.
212      * @protected
213      */
214     this.fillCorners = function (overwrite) {
215         var H = this.item;
216         if (overwrite || (0 > H[0][0])) {
217             H[0][0] = p_randomMinMax(0, p_pitch);
218         }
219         if (overwrite || (0 > H[p_side - 1][0])) {
220             H[p_side - 1][0] = p_randomMinMax(0, p_pitch);
221         }
222         if (overwrite || (0 > H[p_side - 1][p_side - 1])) {
223             H[p_side - 1][p_side - 1] = p_randomMinMax(0, p_pitch);
224         }
225         if (overwrite || (0 > H[0][p_side - 1])) {
226             H[0][p_side - 1] = p_randomMinMax(0, p_pitch);
227         }
228     };
229 
230     /**
231      * Initialize each border of the map with the specified value.
232      * <br />
233      * The item array should be initialized.
234      * @param {number} value Default height to be applied.
235      * @param {number} borderwidth Width of the border to apply the value.
236      * @protected
237      */
238     this.fillBorders = function (value, borderwidth) {
239         var H = this.item, s = p_side - 1, x, y;
240         for (x = 0; x < p_side; x += 1) {
241             for (y = 0; y < p_side; y += 1) {
242                 if ((x < borderwidth) || (y < borderwidth)|| (x > (width - borderwidth)) || (y > (height - borderwidth))) {
243                   H[x][y] = value;
244                 }
245             }
246         }
247     };
248     
249     /**
250      * Return a value added with a random delta.
251      * <br/>
252      * If the delta is more than 0, it adds to "avg" a random value between "-delta" and "+delta".
253      * <br />
254      * In fact it sets the height between two points. The closer there are, the lower the delta will be.
255      * The function ensure that the result is positive and less than the pitch (max height).
256      * <br />
257      * @param {number} avg
258      * @param {number} delta
259      * @return {number}
260      * @private
261      */
262     function p_addDelta(avg, delta) {
263         if (delta) avg += ((((delta * 2) + 1) * rand()) - delta);
264         return (p_pitch < avg) ? p_pitch : (0 > avg) ? 0 : floor(avg);
265     }
266     
267     /**
268      * Generate a random map.
269      * <br />
270      * Parameters indicates "top/left", "right/bottom" limits, and "middle" coordinates.
271      * <br />
272      * Caution : this function only works with 2^n squares, as it goes recursively with divide by 2.
273      * @param {number} x1 x coordinate of the top 
274      * @param {number} y1
275      * @param {number} x2
276      * @param {number} y2
277      * @param {number} xm
278      * @param {number} ym
279      * @protected
280      */
281     this.make = function (x1, y1, x2, y2, xm, ym) {
282         // Notice : Removing floor below could produce more realistics maps (it adds more noise).
283                  // Should it be set by default ? Next "floor" is done in p_addDelta().
284                  // (caution : it takes more time on FF when removed).
285         var delta = floor((x2 - xm) / p_ratio),
286             xmym = 0, H = this.item,
287             x1y1  = H[x1][y1],
288             x2y2  = H[x2][y2],
289             x1y2  = H[x1][y2],
290             x2y1  = H[x2][y1];
291 
292         // Set a random height for the middle of the current square.
293         // p_addDelta is called with the average height of the 4 points.
294         if (0 > H[xm][ym]) H[xm][ym] = p_addDelta((x1y1 + x1y2 + x2y2 + x2y1) / 4, delta);
295         xmym = H[xm][ym];
296   
297         // Set a random height for the middle of the hypotenuse of each triangle.
298         // p_addDelta is called with the average height of the 3 points.
299         if (0 > H[xm][y1]) H[xm][y1] = p_addDelta((x1y1 + x2y1 + xmym) / 3, delta);
300         if (0 > H[xm][y2]) H[xm][y2] = p_addDelta((x1y2 + x2y2 + xmym) / 3, delta);
301         if (0 > H[x2][ym]) H[x2][ym] = p_addDelta((x2y1 + x2y2 + xmym) / 3, delta);
302         if (0 > H[x1][ym]) H[x1][ym] = p_addDelta((x1y1 + x1y2 + xmym) / 3, delta);
303 
304         // Go on if there's space left to fill between points.
305         if (((x2 - x1) > 2) || ((y2 - y1) > 2)) {
306             delta = (xm - x1) / 2;
307             this.make(xm, ym, x2, y2, xm + delta, ym + delta);
308             this.make(x1, ym, xm, y2, xm - delta, ym + delta);
309             this.make(x1, y1, xm, ym, xm - delta, ym - delta);
310             this.make(xm, y1, x2, ym, xm + delta, ym - delta);
311         }
312     };
313 
314     /**
315      * Set cells height to be closer to other adjacent cells.
316      * @protected
317      */
318     this.smooth = function () {
319         var H = this.item, x = 0, y = 0, sum = 0, s = p_side - 1, xm1 = [], xp1 = [], x0 = [];
320         // Goes through every lines and columns except first and last one,
321         // because "sum" below takes every items around the current one.
322         for (xm1 = H[x], x = 1, x0 = H[x]; x < s; x += 1) {
323             xp1 = H[x + 1];
324             for (y = 1, sum = x0[1]; y < s; y += 1) {
325                 sum += (xm1[y - 1] + xm1[y] + xm1[y + 1] + x0[y] + x0[y + 1] + xp1[y - 1] + xp1[y] + xp1[y + 1]);
326                 sum = floor((4 < (sum % 9) ? 1 : 0) + (sum / 9));
327                 if (p_pitch < sum) sum = p_pitch;
328                 x0[y] = sum;
329             }
330             xm1 = x0;
331             x0 = xp1;
332         }
333     };
334 
335     /**
336      * Crop the current map according to specific size.
337      * <br />
338      * The internal "item" array will then be reduced.
339      * @param {number} [opt_width] Width of the cropped map.
340      * @param {number} [opt_height] Height of the cropped map.
341      * @protected
342      */
343     this.crop = function (opt_width, opt_height) {
344         // Initialise width and height if not specified.
345         opt_width  = typeof opt_width  != 'undefined' ? opt_width  : width;
346         opt_height = typeof opt_height != 'undefined' ? opt_height : height;
347         
348         var a = this.item.slice(CROP_SIZE, opt_width + CROP_SIZE),
349             hc = opt_height + CROP_SIZE,
350             x = a.length;
351         while (x--) a[x] = a[x].slice(CROP_SIZE, hc);
352         this.item = a;
353     };
354 
355     /**
356      * Copy data from another map to the current object at x, y coordinates.
357      * @param {array} source Source array that whose values will replace a part of the internal map array.
358      * @param {number} from_x X coordinate of the item array to start the copy
359      * @param {number} from_y Y coordinate of the item array to start the copy
360      * @protected
361      */
362     this.copy = function (source, from_x, from_y) {
363         var D = this.item, S = source.item, x, y;
364         for (x = 0; x < S.length; x += 1) 
365             for (y = 0; y < S[0].length ; y += 1)
366                 D[from_x + x][from_y + y] = S[x][y];
367     };
368     
369     /**
370      * This is a shortcut to call function make() with default parameters.
371      * <br />
372      * All necessary variables should have been initialized.
373      * <br />
374      * The "item" array will be set to "-1" if not already done.
375      * @protected
376      */
377     this.makeMap = function () {
378         // Initialize the "item" array with default height to -1
379         // and corners set to random values.
380         if (this.item.length < 1) {
381           this.initialize(-1);
382           this.fillCorners(true);
383         }
384         // Do map!
385         this.make(0, 0, p_side - 1, p_side - 1, floor((p_side - 1) / 2), floor((p_side - 1) / 2));
386     };
387     
388     /**
389      * Allow you to set a seed used to regenerate identical maps over multiple calls.
390      * @param {*} [opt_seed] Seed used to generate the current map.
391      * @protected
392      */
393     this.setAleaSeed = function (opt_seed) {
394         // Use Alea() if exists.
395         if (typeof Alea != 'undefined') {
396             if ((typeof opt_seed == 'undefined') || (opt_seed == null)) {
397                 rand = Alea();
398             } else {
399                 // Use "opt_seed" if any.
400                 rand = Alea(opt_seed);
401             }
402         }
403     };
404     
405     /**
406      * Build a map (size is 0-based).
407      * <br />
408      * doMap(5, 10) will return a map with dimension [0 .. 4][0 .. 9]
409      * but as we need 2 points to make a cell we'll have 4x9 cells (= 36 true cells displayed).
410      * @example
411      * var oMap = new HeightMap(99, 0.54);
412      * oMap.doMap(14, 34);
413      * @param {number} p_width
414      * @param {number} p_height
415      * @param {boolean} [bInitialize=true] Tell to the function to goes through an initialization of the map array.
416      *                              Then will call "SetSide", "initialize" and "fillcorners" with default values.
417      * @protected
418      */
419     this.doMap = function (p_width, p_height, bInitialize) {
420     
421         if ((typeof bInitialize == 'undefined') || (bInitialize === null) || (bInitialize === true)) {
422             // Initialize side width.
423             this.setSide(p_width, p_height);
424             
425             // To test speed, uncomment line below :
426             // if (isdebug) dbg_date[2] = new Date();
427             
428             // Initialize height.
429             this.initialize(-1);
430             // Initialize corners.
431             this.fillCorners(true);
432         }
433         // To test speed, uncomment line below :
434         // if (isdebug) dbg_date[3] = new Date();
435 
436         // Do map!
437         this.make(0, 0, p_side - 1, p_side - 1, floor((p_side - 1) / 2), floor((p_side - 1) / 2));
438         
439         // To test speed, uncomment line below :
440         // if (isdebug) dbg_date[4] = new Date();
441  
442         // Smooth height map to remove weird points.
443         this.smooth();
444         
445         // To test speed, uncomment line below :
446         // if (isdebug) dbg_date[5] = new Date();
447 
448         // Crop the working map to get the requested map.
449         this.crop();
450 
451         // To test speed, uncomment line below :
452         // if (isdebug) dbg_date[6] = new Date();
453     };
454     
455     /**
456      * Calculate the side of the "square" used to generate the random heightmap.
457      * @param {number} [opt_width]
458      * @param {number} [opt_height]
459      * @return {number} calculated side to be used
460      * @protected
461      */
462     this.setSide = function (opt_width, opt_height) {
463         // Default values if none provided.
464         // It also limits size to [MAX_SIDE]x[MAX_SIDE], to avoid big generation time.
465         if ((opt_width !== undefined) && (opt_width !== null) && (!isNaN(opt_width)) && (opt_width > MIN_SIDE) && (opt_width < MAX_SIDE)) {
466             width = floor(opt_width);
467         } else {
468             width = 127;
469         }
470         if ((opt_height !== undefined) && (opt_height !== null) && (!isNaN(opt_height)) && (opt_height > MIN_SIDE) && (opt_height < MAX_SIDE)) {
471             height = floor(opt_height);
472         } else {
473             height = 127;
474         }
475         // We'll exclude all the border lines to avoid weird point,
476         // so we enlarge the map size with (width+2) and (height+2).
477         opt_width += (2 * CROP_SIZE);
478         opt_height += (2 * CROP_SIZE);
479 
480         // Core engine (function make) works only with squares.
481         // (BTW, we also have better results with 2^N x 2^N maps)
482         // Let's keep the bigger side.
483         p_side = Math.max(opt_width, opt_height);
484 
485         // Look for closest 2^n size (better results from 2^7).
486         var n = 4;
487         while (Math.pow(2, n) < p_side) {
488             n++;
489         }
490         // At this stage, working size will be ((Math.pow(2, n) + 1) x (Math.pow(2, n) + 1))
491         p_side = Math.pow(2, n) + 1;
492         
493         // Return calculated value.
494         return p_side;
495     };
496     
497     if ((opt_pitch !== undefined) && (opt_pitch !== null) && (!isNaN(opt_pitch))) {
498         p_pitch = floor(opt_pitch); // Just in case, pitch is floored, need to be an integer.
499     }
500     if ((opt_ratio !== undefined) && (opt_ratio !== null) && (!isNaN(opt_ratio))) {
501         p_ratio = opt_ratio;
502     }
503     // Set width and height and calculate side for the internal square
504     this.setSide(opt_width, opt_height);
505 
506     // Use Alea() if exists.
507     if (typeof Alea == 'undefined') {
508         rand = Math.random;
509     } else {
510         rand = Alea();
511     }
512 }
