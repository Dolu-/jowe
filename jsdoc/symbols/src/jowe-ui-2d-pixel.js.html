 30 /**
 31  * @fileOverview This file contains the jowe_ui_2d_pixel object (<a href="http://code.google.com/p/jowe/">jOWE</a>).
 32  * @author Ludovic Lefebvre
 33  * @version 1.0
 34  */
 35 
 36 /**
 37  * Draw a map from a given heightmap object in a canvas tag.
 38  * @class Draws a map with each item in the heightmap array being a pixel.
 39  * @see How to use the <a href="http://jowe.ouebfrance.com/examples-jowe-ui-2d-pixel.html">jowe_ui_2d_pixel</a> object?
 40  * @param {string} canvas_id          Id of the canvas tag to be used to draw the map.
 41  * @param {number} canvas_width       Width of the canvas.
 42  * @param {number} canvas_height      Height of the canvas.
 43  * @param {string} [canvas_backcolor="#000"] Background color to be used when drawing the canvas.
 44  */ 
 45 function jowe_ui_2d_pixel(canvas_id, canvas_width, canvas_height, canvas_backcolor) {
 46     "use strict";
 47     
 48     var width,
 49         height,
 50         backcolor = '#000',
 51         canvas_map;
 52 
 53     // Handler to the canvas object used to display the map.
 54     this.map = null;
 55     
 56     if ((canvas_width !== undefined) && (canvas_width !== null) && (!isNaN(canvas_width))) {
 57         width = canvas_width;
 58     } else {
 59         return false;
 60     }
 61     if ((canvas_height !== undefined) && (canvas_height !== null) && (!isNaN(canvas_height))) {
 62         height = canvas_height;
 63     } else {
 64         return false;
 65     }
 66     if ((canvas_backcolor !== undefined) && (canvas_backcolor !== null)) {
 67         backcolor = canvas_backcolor;
 68     }
 69 
 70     canvas_map = document.getElementById(canvas_id);
 71 
 72     // Test for canvas availability.
 73     if (canvas_map.getContext) {
 74         canvas_map.width = width;
 75         canvas_map.height = height;
 76         this.map = canvas_map.getContext('2d');
 77         this.map.fillStyle = backcolor;
 78         this.map.fillRect(0, 0, width, height);
 79     }
 80 
 81     /**
 82      * Draw a pixel map.
 83      * <br />
 84      * The function generates a true image from the heightmap. Each item becomes a pixel.
 85      * <br />
 86      * Even if it's possible to zoom in, it's not recommended as it will produce blurred map.
 87      * @param {array} items Heightmap to display
 88      * @param {number} angle To rotate the map (in degree).
 89      * @param {number} left Left offset from which starting display
 90      * @param {number} top Top offset from which starting display
 91      * @param {array} colors Array of colors associated to heightmap.
 92      * @param {number} [zoom=1] Zoom factor (from 0.1 to 4).
 93      * @public
 94      */
 95     this.draw = function (items, angle, left, top, colors, zoom) {
 96         
 97         var // dimension of the heightmap.
 98             xc = items.length,
 99             yc = items[0].length,
100             // used for "for" loops.
101             x, y, i = 0, color,
102             // Create temporary/working canvas.
103             tmpCanvas = document.createElement('canvas');
104 
105         if ((zoom == undefined) || (zoom == null) || (isNaN(zoom)) || (zoom < 0) || (zoom > 4)) {
106           zoom = 1;
107         }
108             
109         // Define dimension of the canvas.
110         tmpCanvas.width = xc;
111         tmpCanvas.height = yc;
112         // Initialize the context and image data.
113         var tmpContext = tmpCanvas.getContext('2d'),
114             imgmap = tmpContext.createImageData(xc, yc),
115             pixel = imgmap.data;
116         // Colorize each pixel (every 4 values : R,G,B,A)
117         for (y = 0; y < yc; y += 1) {
118             for (x = 0; x < xc; x += 1) {
119                 color = colors[items[x][y]];
120                 pixel[i++] = parseInt(color.substr(1, 2), 16);
121                 pixel[i++] = parseInt(color.substr(3, 2), 16);
122                 pixel[i++] = parseInt(color.substr(5, 2), 16);
123                 pixel[i++] = 254; // alpha component.
124             }
125         }
126         tmpContext.putImageData(imgmap, 0, 0);
127 
128         // Reset the canvas (fill the background with default color).
129         this.map.width = this.map.width;
130         this.map.fillStyle = backcolor;
131         this.map.fillRect(0, 0, width, height);
132 
133         // Save the current context before transformations.
134         this.map.save();
135                 
136         // [step 1]
137         // Going to the relative position that define the center of the map.
138         this.map.translate((width / 2) + left, (height / 2) + top);
139         // [step 2]
140         // Rotation according to the user choice.
141         // Beware, that's define the pivot point to the center of the map,
142         // not to the current center of the canvas.
143         // If we want to rotate from the center of the canvas, the offset
144         // coordinate needs to be processed accordingly.
145         this.map.rotate(-angle * Math.PI/180);
146         // [step 3]
147         // At last we go to the position of the first tile to be drawn.
148         this.map.translate(- xc * zoom / 2, - yc * zoom / 2);
149         
150         // Display image on canvas.
151         this.map.drawImage(tmpCanvas, 0, 0, xc * zoom, yc * zoom);
152         
153         // Restores the context.
154         this.map.restore();
155     };
156    
157     return (this.map != null);
158 }
