 54 "use strict";
 55 
 56 /**
 57  * @fileOverview This file contains the CityMap object (<a href="https://github.com/Dolu-/jowe">jOWE</a>).
 58  * @author Ludovic Lefebvre
 59  * @version 1.0
 60  */

 62 /**
 63  * Creates a new random city map object.<br />
 64  * A "City" is composed of 5 HeightMap objects :<br />
 65  *   - height<br />
 66  *   - fertility<br />
 67  *   - rainfall<br />
 68  *   - temperature<br />
 69  *   - population<br />
 70  * <br />
 71  * The object represents a large area that contains the city.<br />
 72  * Beware, if you request large city map, it will required 4x the size in memory as it will generate 4 heightmap with the same size.<br />
 73  * 
 74  * @class Creates an map that represents a city area.
 75  * @see How to use the <a href="http://jowe.dolu.fr/examples-jowe-ui-city.html">CityMap</a> object?
 76  * @param {number} [opt_width=100]  Width of the map to be generated (between 50 and 500).
 77  * @param {number} [opt_height=100] Height of the map to be generated (between 50 and 500).
 78  */ 
 79 function CityMap(opt_width, opt_height) {
 80 
 81   var rand;
 82 
 83   /**
 84    * Generate a random value between min and max (both included), result is floored.
 85    * @param {number} min
 86    * @param {number} max
 87    * @return {number} Random value.
 88    * @private
 89    */
 90   function p_randomMinMax(min, max) {
 91     return Math.floor((rand() * ((max - min) + 1)) + min);
 92   }
 93 
 94   /**
 95    * Set seed for every heightmap that will be used in the doCityMap process.
 96    * @param {number} height Seed to be used for height map.
 97    * @param {number} [fertility=height] Seed to be used for fertility map.
 98    * @param {number} [rainfall=height+1] Seed to be used for rainfall map.
 99    * @param {number} [temperature=height] Seed to be used for temperature map.
100    * @param {number} [city=height] Seed to be used for city map.
101    * @protected
102    */
103   this.setSeeds = function (height, fertility, rainfall, temperature, city) {
104     this.height_seed = height;
105     
106     this.fertility_seed   = (((fertility !== undefined) && (fertility !== null) && (!isNaN(fertility))) ? fertility : this.height_seed);
107     this.rainfall_seed    = (((rainfall !== undefined) && (rainfall !== null) && (!isNaN(rainfall))) ? rainfall : this.height_seed + 1);
108     this.temperature_seed = (((temperature !== undefined) && (temperature !== null) && (!isNaN(temperature))) ? temperature : this.height_seed);
109     this.city_seed        = (((city !== undefined) && (city !== null) && (!isNaN(city))) ? city : this.height_seed);
110     
111     // Use Alea() if exists.
112     if (typeof Alea == 'undefined') {
113       rand = Math.random;
114     } else {
115       rand = Alea(height);
116     }
117   }
118   
119   /**
120    * Generate a random city map.
121    * @protected
122    */
123   this.doCityMap = function () {
124     // City consists of 5 arrays : Elevation, Fertility, Rainfall, Temperature, Population.
125     var x, y,
126         elevation,
127         fertility,
128         rainfall,
129         temperature,
130         population;
131     
132     /********************************************************************************
133      * STEP 1 :
134      * Elevation (height) process.
135      * Standard generation, no 
136      */
137     elevation = new HeightMap(this.height_pitch, this.height_ratio, width, height);
138     elevation.setAleaSeed(this.height_seed);
139     // makeMap() includes calls to initialize() and fillCorners().
140     elevation.makeMap();
141     elevation.smooth();
142     elevation.crop();
143     /**
144      * End of Elevation (height) process.
145     /********************************************************************************/
146     
147 
148     /********************************************************************************
149      * STEP 2 :
150      * Fertility process. Soil fertility goes from 0 (bad/none) to 100 (excellent).
151      */
152     fertility = new HeightMap(this.fertility_pitch, this.fertility_ratio, width, height);
153     fertility.setAleaSeed(this.fertility_seed);
154     fertility.makeMap();
155     fertility.smooth();
156     fertility.crop();
157     // Fertility : Post processing, remove fertility for ocean, see and snow.
158     for(x = 0; x < width; x += 1) {
159       for(y = 0; y < height; y += 1) {
160         if ((elevation.item[x][y] <= this.fertility_height_min) || (elevation.item[x][y] >= this.fertility_height_max)) {
161           fertility.item[x][y] = 0;
162         }
163       }
164     }
165     /**
166      * End of Fertility process.
167     /********************************************************************************/
168 
169     
170     /********************************************************************************
171      * STEP 3 :
172      * Rainfall process. Rainfall goes from 0 (none) to 20 (very often)
173      */
174     rainfall = new HeightMap(this.rainfall_pitch, this.rainfall_ratio, width, height);
175     rainfall.setAleaSeed(this.rainfall_seed);
176     rainfall.makeMap();
177     // Rainfall : Post processing, but before smooth and crop, snow areas should have reduced rainfall.
178     for(x = 0; x < width; x += 1) {
179       for(y = 0; y < height; y += 1) {
180         if (elevation.item[x][y] >= this.rainfall_height_max) {
181           rainfall.item[x][y] = rainfall.item[x][y] % 5;
182         }
183       }
184     }
185     rainfall.smooth();
186     rainfall.crop();
187     /**
188      * End of Rainfall process.
189     /********************************************************************************/
190 
191     
192     /********************************************************************************
193      * STEP 4 :
194      * Temperature process. Average temperature goes from 0 (5�C) to 20 (25�C)
195      * [0 => 5�C, 5 => 10�C, 10 => 15�C, 15 => 20�C, 20 => 25�C]
196      * This could be changed later, user could decide to move up or down the limits.
197      */
198     temperature = new HeightMap(this.temperature_pitch, this.temperature_ratio, width, height);
199     temperature.setAleaSeed(this.temperature_seed);
200     temperature.makeMap();
201     // Temperature : Post processing, snow areas should have reduced temperature.
202     for(x = 0; x < width; x += 1) {
203       for(y = 0; y < height; y += 1) {
204         if (elevation.item[x][y] >= this.temperature_height_max) {
205           temperature.item[x][y] = temperature.item[x][y] % 10;
206         } else if (temperature.item[x][y] < 5) {
207           temperature.item[x][y] += 1;
208         }
209       }
210     }
211     temperature.smooth();
212     temperature.crop();
213     /**
214      * End of Temperature process.
215     /********************************************************************************/
216 
217     
218     /********************************************************************************
219      * STEP 5 :
220      * Forest/Desert process.
221      */
222     // How to set desert or forest ?
223     // a) internal process
224     // Processed in the same heightmap. Desert and forest are both opposite.
225     // With an heightmap of 7 (0,1,2 => desert) (3,4 => nothing) (5,6,7 => forest)
226     // b) internal or external process
227     // According to the level of temperature, rainfall, fertility and height,
228     // we determine which part of the map could be a forest or a desert.
229     // while drawing the map - let the user decide, or preprocess to avoid recurrent calculations.
230     //
231     // Desert from 0% to 100% => Fertility   from 10 to  0
232     // Desert from 0% to 100% => Temperature from 17 to 20
233     // Desert from 0% to 100% => Rainfall    from  3 to  0
234     // Desert area            => Height      from  2 to  6
235     //
236     // Forest from 0% to 100% to 0% => Fertility   from 55 to 65 to 75
237     // Forest from 0% to 100% to 0% => Temperature from 16 to 18 to 20
238     // Forest from 0% to 100% to 0% => Rainfall    from 16 to 18 to 20
239     // Forest area                  => Height      from  2 to  5
240     /**
241      * End of Fertility process.
242     /********************************************************************************/
243     
244 
245     /********************************************************************************
246      * STEP 6 :
247      * Population process. Population goes from 0 to 100 (eq. 1000 people/km�).
248      * City size is user-defined and fit in the current map.
249      */
250     // Temporary heightmap object used to generate a populated area.
251     var city = new HeightMap(this.city_pitch, this.city_ratio, this.city_width, this.city_height);
252     city.setAleaSeed(this.city_seed);
253     city.initialize(-1);
254     city.fillBorders(0, 4);
255     city.doMap(this.city_width, this.city_height, false);
256     // Put the temporary object in final array with the right size.
257     // Create an empty heightmap with the required size.
258     population = new  HeightMap(0, 0, width, height);
259     population.initialize(0);
260     // Generate random location for the city.
261     var h = 0, w2 = ~~(this.city_width/2), h2 = ~~(this.city_height / 2);
262     while ((h < this.city_height_min) || (h > 6)) {
263       x = p_randomMinMax(5, width  - this.city_width  - 5);
264       y = p_randomMinMax(5, height - this.city_height - 5);
265       h = elevation.item[x + w2][y + h2];
266     }
267     // Copy the temporary city array into the population heightmap.
268     population.copy(city, x, y);
269     // Crop to the requested size.
270     population.crop(width, height)
271     // Population : Post processing, reduce or increase the total population to match approximatively the requested amount.
272     
273     /**
274      * End of Temperature process.
275      * @ignore
276     /********************************************************************************/
277 
278     this.average_height = 0;
279     this.average_fertility = 0;
280     this.average_rainfall = 0;
281     this.average_temperature = 0;
282     this.total_population = 0;
283 
284     this.map = {
285                 height: [],
286                 fertility: [],
287                 rainfall: [],
288                 temperature: [],
289                 population: []
290                };
291     for(x = 0; x < width; x += 1) {
292       this.map.height[x]       = [];
293       this.map.fertility[x]    = [];
294       this.map.rainfall[x]     = [];
295       this.map.temperature[x]  = [];
296       this.map.population[x]   = [];
297       for(y = 0; y < height; y += 1) {
298         this.map.height[x][y]       = elevation.item[x][y];
299         this.map.fertility[x][y]    = fertility.item[x][y];
300         this.map.rainfall[x][y]     = rainfall.item[x][y];
301         this.map.temperature[x][y]  = temperature.item[x][y];
302         if (this.city_height_min <= elevation.item[x][y]) {
303           this.map.population[x][y] = population.item[x][y];
304           this.total_population    += population.item[x][y];
305         } else {
306           this.map.population[x][y] = 0;
307         }
308         this.average_height       += elevation.item[x][y];
309         this.average_fertility    += fertility.item[x][y];
310         this.average_rainfall     += rainfall.item[x][y];
311         this.average_temperature  += temperature.item[x][y];
312       }
313     }
314     
315     // Calculate values for current city.
316     this.average_height      /= (width * height);
317     this.average_fertility   /= (width * height);
318     this.average_rainfall    /= (width * height);
319     this.average_temperature /= (width * height);
320     this.total_population    *= 100;
321     
322     // this.ratio = {ocean : 0, sea : 0, plain : 0, moutain : 0, snow : 0};
323   }
324 
325   if ((opt_width !== undefined) && (opt_width !== null) && (!isNaN(opt_width)) && (opt_width >= 50) && (opt_width <= 500)) {
326     width = opt_width;
327   } else {
328     width = 100;
329   }
330   
331   if ((opt_height !== undefined) && (opt_height !== null) && (!isNaN(opt_height)) && (opt_height >= 50) && (opt_height <= 500)) {
332     height = opt_height;
333   } else {
334     height = 100;
335   }
336   
337   // Default values for heightmaps.
338   this.height_pitch = 8;
339   this.height_ratio = 3.1;
340   
341   this.fertility_pitch = 100;
342   this.fertility_ratio = 0.1;
343   this.fertility_height_min = 2;
344   this.fertility_height_max = 8;
345 
346   this.rainfall_pitch = 20;
347   this.rainfall_ratio = 3.9;
348   this.rainfall_height_max = 8;
349   
350   this.temperature_pitch = 20;
351   this.temperature_ratio = 5;
352   this.temperature_height_max = 7;
353 
354   this.city_pitch = 100;
355   this.city_ratio = 0.1;
356   this.city_width = 30
357   this.city_height = 30;
358   this.city_height_min = 3;
359   
360   this.setSeeds(1000);
361 }
