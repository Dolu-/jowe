 36  * @author Ludovic Lefebvre

 40 /**
 41  * Draws an hexagonal map from a given heightmap object in a canvas tag.
 42  * @class Draws an map with hexagonal tiles.
 43  * @see How to use the <a href="http://jowe.dolu.fr/examples-jowe-ui-2d-hexa.html">jowe_ui_2d_hexa</a> object?
 44  * @param {string} canvas_id          Id of the canvas tag to be used to draw the hexagonal map.
 45  * @param {number} canvas_width       Width of the canvas.
 46  * @param {number} canvas_height      Height of the canvas.
 47  * @param {string} [canvas_backcolor="#000"] Background color to be used when drawing the canvas.
 48  */ 
 49 function jowe_ui_2d_hexa(canvas_id, canvas_width, canvas_height, canvas_backcolor) {
 50     "use strict";
 51     
 52     var width,
 53         height,
 54         backcolor = '#000',
 55         canvas_map;
 56 
 57     // Handler to the canvas object used to display the map.
 58     this.map = null;
 59     
 60     //
 61     this.cursor = {
 62                     visible: false,
 63                     x: -1,
 64                     y: -1,
 65                     mX: -1,
 66                     mY: -1,
 67                     color: "#ff0000"
 68                   };
 69         
 70     if ((canvas_width !== undefined) && (canvas_width !== null) && (!isNaN(canvas_width))) {
 71         width = canvas_width;
 72     } else {
 73         return false;
 74     }
 75     if ((canvas_height !== undefined) && (canvas_height !== null) && (!isNaN(canvas_height))) {
 76         height = canvas_height;
 77     } else {
 78         return false;
 79     }
 80     if ((canvas_backcolor !== undefined) && (canvas_backcolor !== null)) {
 81         backcolor = canvas_backcolor;
 82     }
 83 
 84     canvas_map = document.getElementById(canvas_id);
 85 
 86     // Test for canvas availability.
 87     if (canvas_map.getContext) {
 88         canvas_map.width = width;
 89         canvas_map.height = height;
 90         this.map = canvas_map.getContext('2d');
 91         this.map.fillStyle = backcolor;
 92         this.map.fillRect(0, 0, width, height);
 93     }
 94 
 95     /**
 96      * This function draws an hexagonal map in the canvas.
 97      * @param {array} items Array of vertices in the polygon.
 98      * @param {number} angle Number containing the x-coordinates of the polygon's vertices.
 99      * @param {number} x_off X-coordinate of the offset point.
100      * @param {number} x_off Y-coordinate of the offset point.
101      * @param {number} height Height of the hexagonal tiles.
102      * @param {number} width Width of the hexagonal tiles.
103      * @param {array} colors Array of color used to draw the map.
104      * @param {boolean} [alt=false] To be set to True for straight map (default is False - skew map).
105      * @param {number} [border=0.4] Defines the space between tiles.
106      * @protected
107      */
108     this.draw = function (items, angle, x_off, y_off, h, w, colors, alt, border) {
109 
110         var // used for "for" loops.
111             x, y,
112             // temporarly x,y coordinates.
113             tx, ty,
114             mtx, mty,
115             wdiv2 = w/2,
116             hdiv2 = h/2,
117             hdiv4 = h/4,
118             // distance between tiles.
119             dist = 0.5,
120             // dimension of the heightmap.
121             xc = items.length,
122             yc = items[0].length,
123             // corners of the drawing zone :
124             // pt#[0] = top left
125             // pt#[1] = top right
126             // pt#[2] = bottom right
127             // pt#[3] = bottom left
128             ptx = [], pty = [];
129 
130         if ((alt == undefined) || (alt == null) || (alt != true)) {
131           alt = false;
132         }
133         if ((border !== undefined) && (border !== null) && (!isNaN(border))) {
134             dist = border;
135         }
136             
137         // Initialize the canvas.
138         this.map.fillStyle = backcolor;
139         this.map.fillRect(0, 0, width, height);
140         this.map.width = 1;
141         
142         // Save the current context before transformations.
143         this.map.save();
144 
145         // [step 1]
146         // Going to the relative position that define the center of the map.
147         this.map.translate((width / 2) + x_off, (height / 2) + y_off);
148         // [step 2]
149         // Rotation according to the user choice.
150         // Beware, that's define the pivot point to the center of the map,
151         // not to the current center of the canvas.
152         // If we want to rotate from the center of the canvas, the offset
153         // coordinate needs to be processed accordingly.
154         this.map.rotate(-angle * Math.PI/180);
155         // [step 3]
156         // At last we go in the center of the current map.
157         if (alt == false) {
158           mtx = - (((xc * (w + dist)) - (    wdiv2)) / 2)  + (((w + dist) / 2) * (yc / 2));
159         } else {
160           mtx = -  ((xc * (w + dist)) - (3 * wdiv2)) / 2;
161         }
162         mty = - ((yc * ((3 * hdiv4) + dist)) - (3 * hdiv4)) / 2;
163         this.map.translate(mtx, mty);
164         
165         //
166         // TODO : build a smaller rectangle than the canvas and check that at least one of its corner is included
167         //        in the rectangle of the map, if not return false.
168         //
169         
170         // Coordinates of the corners of the drawing area.
171         // Each tile that's not inside this area won't be drawn.
172         ptx[0] = ptx[3] = -(width  / 2) - x_off - (w + dist);
173         pty[0] = pty[1] = -(height / 2) - y_off - (h + dist);
174         ptx[1] = ptx[2] =  (width  / 2) - x_off + (w + dist);
175         pty[2] = pty[3] =  (height / 2) - y_off + (h + dist);
176         // Loop to take in account the rotation and the last offset.
177         for (var n = 0; n < 4; n += 1) {
178             // Rotate the point coordinate.
179             tx = Math.floor((Math.cos(angle * Math.PI/180) * ptx[n]) - (Math.sin(angle * Math.PI/180) * pty[n]));
180             ty = Math.floor((Math.sin(angle * Math.PI/180) * ptx[n]) + (Math.cos(angle * Math.PI/180) * pty[n]));
181             // Then add the dimension of the heightmap according to the last translation (step 3)
182             ptx[n] = tx - mtx;
183             pty[n] = ty - mty;
184             
185             /* [uncomment for dev only :]
186             this.map.fillStyle = '#00ffff';
187             this.map.fillRect(ptx[n] - 2, pty[n] - 2, 4, 4);
188             */
189         }
190         
191         /* [uncomment for dev only :]
192         this.map.beginPath();
193         this.map.width = 1;
194         this.map.strokeStyle = '#00ffff';
195         this.map.moveTo(ptx[0],pty[0]);
196         this.map.lineTo(ptx[1],pty[1]);
197         this.map.lineTo(ptx[2],pty[2]);
198         this.map.lineTo(ptx[3],pty[3]);
199         this.map.lineTo(ptx[0],pty[0]);
200         this.map.stroke();
201         */
202 
203         //
204         // TODO : simplify (factorize) calculations to speed up the loop.
205         //
206         
207         // Going through each tile.
208         for (y = 0; y < yc; y += 1) {
209             ty = y * (3/4 * h + dist);
210             for (x = 0; x < xc; x += 1) {
211                 // Calculates the "top-left" coordinate of the current tile.
212                 if (alt == false) {
213                   // Skew map.
214                   tx = x * (w + dist) - (y * ((w + dist) / 2));
215                 } else {
216                   // Straight map.
217                   tx = x * (w + dist) - ((y % 2) * ((w + dist) / 2));
218                 }
219                 // Draws the tile only if it's inside the drawing area.
220                 // (original method from http://stackoverflow.com/questions/3990343/3d-drawing-in-canvas-with-htmljs)
221                 if (pnpoly(4, ptx, pty, tx, ty)) {
222                 
223                     fillHexaTile(this.map, colors[items[x][y]], tx, ty, wdiv2, hdiv2, hdiv4);
224                     
225                     if (this.map.isPointInPath(this.cursor.mX, this.cursor.mY)) {
226                       this.cursor.x = x;
227                       this.cursor.y = y;
228                       this.cursor.mX = tx;
229                       this.cursor.mY = ty;
230                     }
231                 }
232             }
233         }
234         
235         if (this.cursor.visible == true) {
236           this.map.lineWidth = 2;
237           strokeHexaTile(this.map, this.cursor.color, this.cursor.mX, this.cursor.mY, wdiv2, hdiv2, hdiv4);
238         }
239         
240         // Restores the context.
241         this.map.restore();
242     };
243     
244     /**
245      * This function draws a filled hexagon.
246      * @param {*} m Canvas context to draw on.
247      * @param {string} c Array containing the x-coordinates of the polygon's vertices.
248      * @param {number} tx X-coordinate of the center of the hexagon.
249      * @param {number} tx X-coordinate of the center of the hexagon.
250      * @param {number} wdiv2 Width divided by 2.
251      * @param {number} hdiv2 Height divided by 2.
252      * @param {number} hdiv4 Height divided by 4.
253      * @private
254      */
255     function fillHexaTile(m, c, tx, ty, wdiv2, hdiv2, hdiv4)
256     {
257         m.beginPath();
258         m.fillStyle = c;
259         m.moveTo(tx        , ty - hdiv2);
260         m.lineTo(tx + wdiv2, ty - hdiv4);
261         m.lineTo(tx + wdiv2, ty + hdiv4);
262         m.lineTo(tx        , ty + hdiv2);
263         m.lineTo(tx - wdiv2, ty + hdiv4);
264         m.lineTo(tx - wdiv2, ty - hdiv4);
265         m.fill();
266     }
267 
268     /**
269      * This function draws an empty hexagon.
270      * @param {*} m Canvas context to draw on.
271      * @param {string} c Array containing the x-coordinates of the polygon's vertices.
272      * @param {number} tx X-coordinate of the center of the hexagon.
273      * @param {number} tx X-coordinate of the center of the hexagon.
274      * @param {number} wdiv2 Width divided by 2.
275      * @param {number} hdiv2 Height divided by 2.
276      * @param {number} hdiv4 Height divided by 4.
277      * @private
278      */
279     function strokeHexaTile(m, c, tx, ty, wdiv2, hdiv2, hdiv4)
280     {
281         m.beginPath();
282         m.strokeStyle = c;
283         m.moveTo(tx        , ty - hdiv2);
284         m.lineTo(tx + wdiv2, ty - hdiv4);
285         m.lineTo(tx + wdiv2, ty + hdiv4);
286         m.lineTo(tx        , ty + hdiv2);
287         m.lineTo(tx - wdiv2, ty + hdiv4);
288         m.lineTo(tx - wdiv2, ty - hdiv4);
289         m.closePath();
290         m.stroke();
291     }
292     
293     /**
294      * This function find if a point lies within a polygon.
295      * @param {number} nvert Number of vertices in the polygon.
296      * @param {array} vertx Array containing the x-coordinates of the polygon's vertices.
297      * @param {array} verty Array containing the y-coordinates of the polygon's vertices.
298      * @param {number} testx X-coordinate of the test point.
299      * @param {number} testy Y-coordinate of the test point.
300      * @return {boolean}
301      * @private
302      */
303     function pnpoly(nvert, vertx, verty, testx, testy) {     
304         var i, j, c = false;
305         for (i = 0, j = nvert - 1; i < nvert; j = i++) {
306             if (((verty[i] > testy) !== (verty[j] > testy)) &&
307                 (testx < (vertx[j] - vertx[i]) * (testy - verty[i]) / (verty[j] - verty[i]) + vertx[i])) {
308                 c = !c;
309             }
310         }
311         return c;
312     }
313     
314     return (this.map != null);
315 }
